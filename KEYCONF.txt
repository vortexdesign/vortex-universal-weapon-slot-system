// VUWS Key Configuration
//
// Slot editor opens via Q, exclude menu via Z (non-destructive defaultbinds)
// Other commands fire netevents handled by VUWS_SlotCommands.NetworkProcess

alias vuws_slot_editor "openmenu VUWS_SlotEditorMenu"
alias vuws_exclude_menu "openmenu VUWS_ExcludeListMenu"

defaultbind q vuws_slot_editor
defaultbind z vuws_exclude_menu

// One-shot binding command for users with existing 1-9 binds
// `vuws_bind_slot_keys` rebinds 0-9 to the VUWS-aware aliases, INI persistent
// `vuws_unbind_slot_keys` reverts
alias vuws_bind_slot_keys "bind 1 vuws_slotk_1; bind 2 vuws_slotk_2; bind 3 vuws_slotk_3; bind 4 vuws_slotk_4; bind 5 vuws_slotk_5; bind 6 vuws_slotk_6; bind 7 vuws_slotk_7; bind 8 vuws_slotk_8; bind 9 vuws_slotk_9; bind 0 vuws_slotk_0; echo VUWS: 0-9 rebound to VUWS-aware slot keys"

alias vuws_unbind_slot_keys "bind 0 \"slot 0\"; bind 1 \"slot 1\"; bind 2 \"slot 2\"; bind 3 \"slot 3\"; bind 4 \"slot 4\"; bind 5 \"slot 5\"; bind 6 \"slot 6\"; bind 7 \"slot 7\"; bind 8 \"slot 8\"; bind 9 \"slot 9\"; echo VUWS: 0-9 reverted to engine slot keys"

// One-shot mwheel rebind for users whose wheel is bound to engine weapnext/weapprev
// `vuws_bind_scroll_keys` rebinds mwheelup/down to the VUWS-aware aliases, INI persistent
// `vuws_unbind_scroll_keys` reverts
alias vuws_bind_scroll_keys "bind mwheelup vuws_weapnext; bind mwheeldown vuws_weapprev; echo VUWS: mouse wheel rebound to VUWS-aware cycle"
alias vuws_unbind_scroll_keys "bind mwheelup weapnext; bind mwheeldown weapprev; echo VUWS: mouse wheel reverted to engine cycle"

// vuws_reset_defaults
//
// Chained `set` commands not netevent
// ZScript CVar.SetInt on user-scope CVars doesn't reliably persist via netevent
// Console `set` path persists, same trick as VUOS / VUSS
//
// Resets all settings CVars but NOT vuws_user_slot_* (those reset via vuws_clear_user_slots)
