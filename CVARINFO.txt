// VUWS CVar Definitions

// behavior CVars server-scope (deterministic gameplay rules across coop clients)
// personal prefs user-scope (caps, slot config, HUD, colors)
// groups: master, caps, behavior, reassign persistence, use-pickup, HUD, colors

// --- Master toggles ---

// --- Behavior ---

// --- Use-key pickup ---

server int vuws_pickup_detection = 2;               // 0=look-trace, 1=proximity, 2=auto (trace, fall back to proximity if no hit)

// --- Auto-show slot editor ---
user bool vuws_auto_show_on_first_map = false;      // pop the menu once per session on first map
user bool vuws_auto_show_on_map_change = false;     // pop on every map change
user bool vuws_auto_show_on_death = false;          // pop on respawn after death

user int vuws_notify_duration = 70;                 // Toast duration in tics
user float vuws_notify_opacity = 0.9;               // Toast opacity 0.0-1.0

user int vuws_color_blocked = 6;                    // Block/reject toast color (6=Red)
user int vuws_color_swap = 5;                       // Swap/bump toast color (5=Gold)
